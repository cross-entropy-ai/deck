# deck Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A single Rust binary that renders a tmux session sidebar using ratatui, with toggle support via `tmux split-window`.

**Architecture:** Single binary with two modes — `deck` (toggle: spawn/kill sidebar pane) and `deck --render` (TUI process inside the pane). Data comes from shelling out to `tmux` and `git` CLI. Polling at 1s interval for updates.

**Tech Stack:** Rust, ratatui 0.30, crossterm 0.29

```rust
use std::process::Command;

/// Info about a tmux session.
#[derive(Debug, Clone)]
pub struct SessionInfo {
    pub name: String,
    pub dir: String,
    pub attached: bool,
    pub window_count: u32,
}

/// Info about a tmux pane.
#[derive(Debug, Clone)]
pub struct PaneInfo {
    pub id: String,
    pub title: String,
    pub session_name: String,
    pub window_id: String,
}

/// Run a tmux command and return stdout, trimmed.
fn tmux(args: &[&str]) -> Option<String> {
    let output = Command::new("tmux")
        .args(args)
        .output()
        .ok()?;
    if !output.status.success() {
        return None;
    }
    Some(String::from_utf8_lossy(&output.stdout).trim().to_string())
}

/// List all tmux sessions.
pub fn list_sessions() -> Vec<SessionInfo> {
    let format = "#{session_name}\t#{session_path}\t#{session_attached}\t#{session_windows}";
    let Some(raw) = tmux(&["list-sessions", "-F", format]) else {
        return Vec::new();
    };
    raw.lines()
        .filter_map(|line| {
            let parts: Vec<&str> = line.split('\t').collect();
            if parts.len() < 4 {
                return None;
            }
            Some(SessionInfo {
                name: parts[0].to_string(),
                dir: parts[1].to_string(),
                attached: parts[2] == "1",
                window_count: parts[3].parse().unwrap_or(0),
            })
        })
        .collect()
}

/// Get the current session name (from the first attached client).
pub fn current_session() -> Option<String> {
    tmux(&["display-message", "-p", "#{session_name}"])
}

/// Get the current window ID.
pub fn current_window_id() -> Option<String> {
    tmux(&["display-message", "-p", "#{window_id}"])
}

/// List all panes across all sessions.
pub fn list_panes() -> Vec<PaneInfo> {
    let format = "#{pane_id}\t#{pane_title}\t#{session_name}\t#{window_id}";
    let Some(raw) = tmux(&["list-panes", "-a", "-F", format]) else {
        return Vec::new();
    };
    raw.lines()
        .filter_map(|line| {
            let parts: Vec<&str> = line.split('\t').collect();
            if parts.len() < 4 {
                return None;
            }
            Some(PaneInfo {
                id: parts[0].to_string(),
                title: parts[1].to_string(),
                session_name: parts[2].to_string(),
                window_id: parts[3].to_string(),
            })
        })
        .collect()
}

/// Switch the current client to a different session.
pub fn switch_session(name: &str) {
    let _ = tmux(&["switch-client", "-t", name]);
}

/// Kill a specific pane.
pub fn kill_pane(pane_id: &str) {
    let _ = tmux(&["kill-pane", "-t", pane_id]);
}

/// Set a pane's title.
pub fn set_pane_title(pane_id: &str, title: &str) {
    let _ = tmux(&["select-pane", "-t", pane_id, "-T", title]);
}

/// Split window and spawn a command, returning the new pane ID.
pub fn split_window(width: u16, command: &str) -> Option<String> {
    tmux(&[
        "split-window", "-hb", "-f",
        "-l", &width.to_string(),
        "-P", "-F", "#{pane_id}",
        command,
    ])
}
```

```rust
/// Git info for a directory.
#[derive(Debug, Clone, Default)]
pub struct GitInfo {
    pub branch: String,
    pub dirty: bool,
}

/// Get git branch and dirty status for a directory.
pub fn get_git_info(dir: &str) -> GitInfo {
    if dir.is_empty() {
        return GitInfo::default();
    }

    let branch = Command::new("git")
        .args(["rev-parse", "--abbrev-ref", "HEAD"])
        .current_dir(dir)
        .output()
        .ok()
        .filter(|o| o.status.success())
        .map(|o| String::from_utf8_lossy(&o.stdout).trim().to_string())
        .unwrap_or_default();

    let dirty = Command::new("git")
        .args(["status", "--porcelain"])
        .current_dir(dir)
        .output()
        .ok()
        .filter(|o| o.status.success())
        .map(|o| !o.stdout.is_empty())
        .unwrap_or(false);

    GitInfo { branch, dirty }
}
```

```rust
use std::time::{Duration, Instant};

use crossterm::event::{self, Event, KeyCode, KeyEventKind};
use ratatui::DefaultTerminal;

use crate::git;
use crate::tmux;
use crate::ui;

/// Data for a single session row.
#[derive(Debug, Clone)]
pub struct SessionRow {
    pub name: String,
    pub dir: String,
    pub branch: String,
    pub dirty: bool,
    pub is_current: bool,
}

pub struct App {
    pub sessions: Vec<SessionRow>,
    pub focused: usize,
    pub current_session: String,
    pub exit: bool,
}

impl App {
    pub fn new() -> Self {
        let mut app = App {
            sessions: Vec::new(),
            focused: 0,
            current_session: String::new(),
            exit: false,
        };
        app.refresh();
        app
    }

    /// Refresh session list from tmux + git.
    pub fn refresh(&mut self) {
        let current = tmux::current_session().unwrap_or_default();
        let sessions = tmux::list_sessions();

        // Filter out sessions whose name starts with '_' (internal, like _os_stash)
        self.sessions = sessions
            .into_iter()
            .filter(|s| !s.name.starts_with('_'))
            .map(|s| {
                let git_info = git::get_git_info(&s.dir);
                SessionRow {
                    is_current: s.name == current,
                    name: s.name,
                    dir: s.dir,
                    branch: git_info.branch,
                    dirty: git_info.dirty,
                }
            })
            .collect();

        self.current_session = current;

        // Clamp focused index
        if !self.sessions.is_empty() && self.focused >= self.sessions.len() {
            self.focused = self.sessions.len() - 1;
        }
    }

    /// Main event loop.
    pub fn run(&mut self, terminal: &mut DefaultTerminal) -> std::io::Result<()> {
        let poll_interval = Duration::from_secs(1);
        let mut last_refresh = Instant::now();

        while !self.exit {
            terminal.draw(|frame| ui::draw(frame, self))?;

            // Poll for events with timeout so we can refresh periodically
            if event::poll(Duration::from_millis(100))? {
                if let Event::Key(key) = event::read()? {
                    if key.kind == KeyEventKind::Press {
                        self.handle_key(key.code);
                    }
                }
            }

            // Periodic refresh
            if last_refresh.elapsed() >= poll_interval {
                self.refresh();
                last_refresh = Instant::now();
            }
        }

        Ok(())
    }

    fn handle_key(&mut self, code: KeyCode) {
        match code {
            KeyCode::Char('q') | KeyCode::Esc => self.exit = true,
            KeyCode::Char('j') | KeyCode::Down => {
                if !self.sessions.is_empty() {
                    self.focused = (self.focused + 1).min(self.sessions.len() - 1);
                }
            }
            KeyCode::Char('k') | KeyCode::Up => {
                if self.focused > 0 {
                    self.focused -= 1;
                }
            }
            KeyCode::Enter => {
                if let Some(session) = self.sessions.get(self.focused) {
                    tmux::switch_session(&session.name);
                    // Update current after switch
                    self.refresh();
                }
            }
            _ => {}
        }
    }
}
```

```rust
use ratatui::layout::{Constraint, Layout, Rect};
use ratatui::style::{Modifier, Style};
use ratatui::text::{Line, Span};
use ratatui::widgets::Paragraph;
use ratatui::Frame;

use crate::app::App;
use crate::theme::*;

pub fn draw(frame: &mut Frame, app: &App) {
    let area = frame.area();

    // Fill background
    frame.render_widget(
        ratatui::widgets::Block::default().style(Style::default().bg(CRUST)),
        area,
    );

    // Layout: header (2 lines) | sessions (fill) | footer (2 lines)
    let [header_area, sessions_area, footer_area] = Layout::vertical([
        Constraint::Length(2),
        Constraint::Min(1),
        Constraint::Length(2),
    ])
    .areas(area);

    draw_header(frame, header_area, app);
    draw_sessions(frame, sessions_area, app);
    draw_footer(frame, footer_area);
}

fn draw_header(frame: &mut Frame, area: Rect, app: &App) {
    let count = app.sessions.len().to_string();
    let line = Line::from(vec![
        Span::raw("  "),
        Span::styled("Sessions", Style::default().fg(SUBTEXT0).add_modifier(Modifier::BOLD)),
        Span::raw(" "),
        Span::styled(count, Style::default().fg(OVERLAY0)),
    ]);
    frame.render_widget(
        Paragraph::new(vec![Line::raw(""), line]).style(Style::default().bg(CRUST)),
        area,
    );
}

fn draw_sessions(frame: &mut Frame, area: Rect, app: &App) {
    if app.sessions.is_empty() {
        let msg = Paragraph::new("  No sessions")
            .style(Style::default().fg(OVERLAY0).bg(CRUST));
        frame.render_widget(msg, area);
        return;
    }

    // Each session card: 3-4 lines + 1 blank = up to 5 lines
    // We need to figure out how many fit and handle scrolling
    let mut lines: Vec<Line> = Vec::new();

    for (i, session) in app.sessions.iter().enumerate() {
        let is_focused = i == app.focused;
        let is_current = session.is_current;

        // Accent bar color
        let accent_color = if is_current {
            GREEN
        } else if is_focused {
            LAVENDER
        } else {
            CRUST // invisible
        };

        let accent = if is_current || is_focused { "▌" } else { " " };
        let name_style = if is_focused || is_current {
            Style::default().fg(TEXT).add_modifier(Modifier::BOLD)
        } else {
            Style::default().fg(SUBTEXT0)
        };
        let index_style = if is_focused {
            Style::default().fg(SUBTEXT0)
        } else {
            Style::default().fg(SURFACE2)
        };

        let bg = if is_focused { SURFACE0 } else { CRUST };

        // Row 1: accent + index + name
        let idx_str = format!("{:>2}", i + 1);
        lines.push(Line::from(vec![
            Span::styled(accent, Style::default().fg(accent_color).bg(bg)),
            Span::styled(" ", Style::default().bg(bg)),
            Span::styled(idx_str, index_style.bg(bg)),
            Span::styled("  ", Style::default().bg(bg)),
            Span::styled(&session.name, name_style.bg(bg)),
        ]));

        // Row 2: directory (shortened)
        let dir_display = shorten_dir(&session.dir);
        let dir_color = if is_focused { TEAL } else { OVERLAY0 };
        lines.push(Line::from(vec![
            Span::styled("      ", Style::default().bg(bg)),
            Span::styled(dir_display, Style::default().fg(dir_color).bg(bg)),
        ]));

        // Row 3: branch + dirty indicator
        if !session.branch.is_empty() {
            let branch_color = if is_focused { PINK } else { OVERLAY0 };
            let mut row3 = vec![
                Span::styled("      ", Style::default().bg(bg)),
                Span::styled(&session.branch, Style::default().fg(branch_color).bg(bg)),
            ];
            if session.dirty {
                row3.push(Span::styled(" ●", Style::default().fg(YELLOW).bg(bg)));
            }
            lines.push(Line::from(row3));
        }

        // Blank line between cards
        lines.push(Line::from(Span::styled(" ", Style::default().bg(CRUST))));
    }

    let paragraph = Paragraph::new(lines)
        .style(Style::default().bg(CRUST))
        .scroll((scroll_offset(app, area.height) as u16, 0));
    frame.render_widget(paragraph, area);
}

fn draw_footer(frame: &mut Frame, area: Rect) {
    let sep = Line::from(Span::styled(
        "─".repeat(area.width as usize),
        Style::default().fg(SURFACE2),
    ));
    let hints = Line::from(vec![
        Span::styled(" j/k", Style::default().fg(OVERLAY0)),
        Span::styled(" nav  ", Style::default().fg(OVERLAY1)),
        Span::styled("⏎", Style::default().fg(OVERLAY0)),
        Span::styled(" switch  ", Style::default().fg(OVERLAY1)),
        Span::styled("q", Style::default().fg(OVERLAY0)),
        Span::styled(" quit", Style::default().fg(OVERLAY1)),
    ]);
    frame.render_widget(
        Paragraph::new(vec![sep, hints]).style(Style::default().bg(CRUST)),
        area,
    );
}

/// Shorten a directory path: replace $HOME with ~, keep last 2 components.
fn shorten_dir(dir: &str) -> String {
    let home = std::env::var("HOME").unwrap_or_default();
    let shortened = if !home.is_empty() && dir.starts_with(&home) {
        format!("~{}", &dir[home.len()..])
    } else {
        dir.to_string()
    };
    shortened
}

/// Calculate scroll offset to keep focused item visible.
fn scroll_offset(app: &App, visible_height: u16) -> usize {
    // Each card is ~4 lines (3 content + 1 blank)
    let card_height = 4;
    let focused_top = app.focused * card_height;
    let focused_bottom = focused_top + card_height;
    let visible = visible_height as usize;

    if focused_bottom > visible {
        focused_bottom - visible
    } else {
        0
    }
}
```

```rust
use crate::tmux;

const SIDEBAR_TITLE: &str = "deck";
const DEFAULT_WIDTH: u16 = 28;

/// Find an existing sidebar pane (by title).
fn find_sidebar_pane() -> Option<String> {
    let panes = tmux::list_panes();
    panes
        .into_iter()
        .find(|p| p.title == SIDEBAR_TITLE)
        .map(|p| p.id)
}

/// Toggle the sidebar: if it exists, kill it; if not, spawn it.
pub fn toggle() {
    if let Some(pane_id) = find_sidebar_pane() {
        // Sidebar exists — kill it
        tmux::kill_pane(&pane_id);
    } else {
        // Sidebar doesn't exist — spawn it
        let exe = std::env::current_exe()
            .map(|p| p.to_string_lossy().to_string())
            .unwrap_or_else(|_| "deck".to_string());

        let command = format!("{} --render", exe);
        if let Some(pane_id) = tmux::split_window(DEFAULT_WIDTH, &command) {
            let pane_id = pane_id.trim().to_string();
            tmux::set_pane_title(&pane_id, SIDEBAR_TITLE);
            // Refocus the original pane (split-window moves focus to new pane)
            // The last pane (right) is the user's working pane
            let _ = std::process::Command::new("tmux")
                .args(["select-pane", "-t", "{right}"])
                .status();
        }
    }
}
```
